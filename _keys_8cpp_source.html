44 S32 FW::keyToUnicode(const String& key)
45 {
46  const char* ptr = key.getPtr();
47  S32 cp = 0;
48 
49  if (*ptr != 'U' || *++ptr != '+')
50  return 0;
51 
52  while (*++ptr)
53  {
54  cp <<= 4;
55 
56  if (*ptr >= '0' && *ptr <= '9')
57  cp += *ptr - '0';
58  else if (*ptr >= 'A' && *ptr <= 'F')
59  cp += *ptr - ('A' - 10);
60  else
61  return 0;
62 
63  if (cp > 0x10FFFF)
64  return 0;
65  }
66 
67  return cp;
68 }
