88 S64 Timer::queryTicks(void)
89 {
90  LARGE_INTEGER ticks;
91  QueryPerformanceCounter(&ticks);
92  ticks.QuadPart = max(s_prevTicks, ticks.QuadPart);
93  s_prevTicks = ticks.QuadPart; // increasing little endian => thread-safe
