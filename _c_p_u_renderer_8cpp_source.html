46 bool CPURenderer::nextBatch(void)
47 {
48  FW_ASSERT(m_scene);
49 
50  // Clean up the previous batch.
51 
52  if (m_batchRays && !m_newBatch)
53  m_batchStart += m_batchRays->getSize();
54  //m_batchRays = NULL;
55 
56  // Generate new batch.
57 
58  U32 randomSeed = (m_enableRandom) ? m_random.getU32() : 0;
59  switch (m_params.rayType)
60  {
61  case RayType_Primary:
62  case RayType_Textured:
63  case RayType_PathTracing:
64  if (!m_newBatch)
65  return false;
66  m_newBatch = false;
67  m_batchRays = &m_primaryRays;
68  break;
69 
70  case RayType_AO:
71  if (!m_raygen.aoCPU(m_secondaryRays, m_primaryRays, *m_scene, m_params.numSamples, m_params.aoRadius, m_newBatch, randomSeed))
72  return false;
73  m_batchRays = &m_secondaryRays;
74  break;
75 
76  case RayType_Diffuse:
77  if (!m_raygen.aoCPU(m_secondaryRays, m_primaryRays, *m_scene, m_params.numSamples, m_cameraFar, m_newBatch, randomSeed))
78  return false;
79  m_secondaryRays.setNeedClosestHit(true);
80  m_batchRays = &m_secondaryRays;
81  break;
82 
83  default:
84  FW_ASSERT(false);
85  return false;
86  }
87 
88  // Sort rays.
89 
90  if (m_params.sortSecondary && !(m_params.rayType == RayType_Primary || m_params.rayType == RayType_Textured || m_params.rayType == RayType_PathTracing))
91  m_batchRays->mortonSort();
92  return true;
93 }

457 Vec4f CPURenderer::getDistanceColor(F32 value, F32 minVal, F32 maxVal)
458 {
459  Vec4f val(1.0f);
460 
461  if (value < minVal) value = minVal;

463 
464  float ratio = (value - minVal)/(maxVal - minVal);
465  // The first color - corresponding to minimum

468  const float minColorBlue = 0.0f;
