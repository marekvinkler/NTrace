98 template <class T> void Deque<T>::set(const T* ptr, int size)
99 {
100  FW_ASSERT(size >= 0);
101 
102  m_items.reset(size);
103  for (int i = 0; i < size; i++)
104  {
105  if (ptr)

107  m_items[i].prev = i - 1;

112  m_last = size - 1;
